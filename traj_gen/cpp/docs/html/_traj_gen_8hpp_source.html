124  template <typename T>
125  T interpolate(const VectorX<T>& xData, const VectorX<T>& yData,const T & x,bool extrapolate){
126  int size = xData.size();
127  int i = 0; // find left end of interval for interpolation
128  if ( x >= xData(size - 2) ) // special case: beyond right end
129  {
130  i = size - 2;
131  }
132  else
133  {
134  while ( x > xData(i+1) ) i++;
135  }
136  T xL = xData(i), yL = yData(i), xR = xData(i+1), yR = yData(i+1); // points on either side (unless beyond ends)
137  if ( !extrapolate ) // if beyond ends of array and not extrapolating
138  {
139  if ( x < xL ) yR = yL;
140  if ( x > xR ) yL = yR;
141  }
142 
143  T dydx = ( yR - yL ) / ( xR - xL ); // gradient
144  return yL + dydx * ( x - xL ); // linear interpolation
145  }

476  VectorX<T> QpBlock<T>::getQpSol(bool & isSolved) {
477  uint nVar = Q.rows();
478  int N_ineq_const = A.rows();
479  int N_eq_const = Aeq.rows();
480  int N_const = N_ineq_const + N_eq_const;
481  // convention follows the qpOASES
482  qpOASES::QProblem qp_obj(nVar, N_const, qpOASES::HST_SEMIDEF);
483  Matrix<double, -1, -1, RowMajor> Hmat(2 * Q.template cast<double>());
484  Matrix<double, 1, -1, RowMajor> gmat(H.template cast<double>());
485  Matrix<double, -1, -1, RowMajor> Amat(N_const, nVar);
486  Amat << MatrixXd(A.template cast<double>()), MatrixXd(Aeq.template cast<double>());
487  Matrix<double, -1, -1, RowMajor> ubAmat(N_const, 1);
488  ubAmat << MatrixXd(b.template cast<double>()), MatrixXd(beq.template cast<double>());
489  MatrixXd MinusInf(N_ineq_const, 1);
490  MinusInf=MinusInf.setConstant(-99999);
491  Matrix<double, -1, -1, RowMajor> lbAmat(N_const, 1);
492  lbAmat << MinusInf, MatrixXd(beq.template cast<double>());
493

495  qpOASES::real_t *gqp = gmat.data();
496  qpOASES::real_t *Aqp = Amat.data();
497  qpOASES::real_t *lbAqp = lbAmat.data();
498  qpOASES::real_t *ubAqp = ubAmat.data();
499 
500  qpOASES::Options options;
501  options.printLevel = qpOASES::PL_LOW;
502  options.terminationTolerance = 1e-10;
503 // options.enableRamping=qpOASES::BT_FALSE;
504  qp_obj.setOptions(options);
505  qpOASES::int_t nWSR = 4000;
506  qp_obj.init(Hqp, gqp, Aqp, NULL, NULL, lbAqp, ubAqp, nWSR);
507  if(qp_obj.isInfeasible()){
508  cout<<"[QP solver] warning: problem is infeasible. "<<endl;
509  }
510 
511  cout << "Acutual nWSR: " << nWSR << " / Original: " << 2000 << endl;
512 
513  qpOASES::real_t xOpt[nVar];
514  qp_obj.getPrimalSolution(xOpt);
515  isSolved = qp_obj.isSolved();
516  if (isSolved)
517  cout<<"[QP solver] Success! "<<endl;
518  else
519 
520  cout<<"[QP solver] Failure. "<<endl;
521  return Map<Matrix<double, -1, 1>>(xOpt, nVar, 1).cast<T>();
522  }

863  template<typename T,size_t dim> QpForm<T,dim> PolyTrajGen<T,dim>::getQPSet() {
864 
865  uint nVar = (this->M)*(N+1);
866  uint nineq = this->getTotalNineq(); // total line of inequality
867  uint neq = this->getTotalNeq();
868  QpForm<T,dim> qpForm(nVar,neq,nineq); // initialize all the container
869 
870  // 1. Objective function
871  spMatrixRow<T> Q(nVar,nVar);
872  for (uint d = 1; d <= this->weight_mask.size(); d++) { // dth order derivative
873  spMatrixRow<T> Qd(nVar, nVar);
874  for (uint m = 0; m < this->M; m++) { // per segment, plug in the Q
875  T dt = this->ts[m + 1] - this->ts[m];
876  sparseBlockCopy<T>(&Qd,MatrixRow<T>(IntDerSquard(d)/pow(dt,2*d-1)),m*(N+1),m*(N+1));
877  }
878  Q += this->weight_mask[d - 1] * Qd;
879  }
880 
881  for (uint dd = 0 ; dd < dim ; dd++)
882  qpForm.qpBlock[dd].Q = Q;
883 
884  uint Idx1 = 0,Idx2 = 0;
885  for (uint m = 0 ; m < this->M ; m++){ // per segment
886  // 2. Ineq constraint for this segment
887  uint Nineq = this->loosePinSet[m].size();
888  ConstraintMatPair<T,dim> Ab = loosePinMatSet((this->loosePinSet[m]).data(),this->loosePinSet[m].size());
889  for (uint dd = 0 ; dd < dim ; dd++) { // per dimension
890  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].A),(Ab.ASet[dd]),Idx1,0);
891  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].b),(Ab.bSet[dd]),Idx1,0);
892  }
893  Idx1+= 2*Nineq;
894 
895  // 3. Eq constraint for this segment
896  ConstraintMatPair<T,dim> AbFix = fixPinMatSet((this->fixPinSet[m]).data(),this->fixPinSet[m].size());
897  ConstraintMatPair<T,dim> AbConti = contiMat(m,seg_state_set[m].Nc-1);
898  uint Neq = seg_state_set[m].getN(), Nf = seg_state_set[m].Nf, Nc = seg_state_set[m].Nc;
899  for (uint dd = 0 ; dd < dim ; dd++) { // inserting A,b
900  // (Aeq)m = ([Afix ; Aconti])m : Aeq*p = beq
901  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].Aeq),(AbFix.ASet[dd]),Idx2,0) ;
902  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].Aeq),(AbConti.ASet[dd]),Idx2+Nf,0);
903  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].beq),(AbFix.bSet[dd]),Idx2,0) ;
904  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].beq),(AbConti.bSet[dd]),Idx2+Nf,0);
905 
906  }
907  Idx2 += Neq;
908  }
909  return qpForm;
910  }
